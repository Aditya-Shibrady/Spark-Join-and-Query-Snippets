val data=sc.textFile("/yelpdatafall/review/review.csv")
val business_file = sc.textFile("/yelpdatafall/business/business.csv")
val values=data.map(_.split("\\^")).map(line=>(line(2),line(3).toFloat)).mapValues(p=>(p,1)).reduceByKey((p,q)=>
(p._1+q._1, p._2+q._2))
val temp = values.map(line=>(line._1,(line._2._1/line._2._2))).sortBy(_._1).sortBy(-_._2).take(10)
val temp2 = sc.parallelize(temp)
val busines = business_file.map(line=>line.split("\\^")).map(line=>(line(0),(line(1),line(2)))).distinct()
val output = temp2.join(busines)
output.foreach(println)
